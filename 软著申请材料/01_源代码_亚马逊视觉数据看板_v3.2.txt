// ========== 三、数据类型自动检测 ==========
function detectDataType(data) {
  if (!data || !data.headers || !data.rows.length) return 'byDate';
  const headers = data.headers.map(h => h.toLowerCase());
  const hasAsin = headers.some(h => h.includes('asin'));
  const hasTitle = headers.some(h => h.includes('title') || h.includes('产品') || h.includes('商品名称'));
  if (hasAsin || hasTitle) return 'byAsin';
  return 'byDate';
}

// ========== 四、数据校验模块 ==========
function validateUploadedData(data, type) {
  const result = { warnings: [], severity: 'warn' };
  if (!data || !data.headers || !data.rows) {
    result.warnings.push('• 文件结构异常：无法识别表头或数据行');
    result.severity = 'error';
    return result;
  }
  const headers = data.headers.map(h => h.toLowerCase());
  if (type === 'business' || type === 'beforeBiz' || type === 'afterBiz' || type === 'yoyLastBiz' || type === 'yoyThisBiz') {
    const hasDate = headers.some(h => h.includes('date') || h.includes('日期'));
    const hasSessions = headers.some(h => h.includes('session') || h.includes('会话'));
    const hasCVR = headers.some(h => h.includes('unit session') || h.includes('转化率'));
    const hasSales = headers.some(h => h.includes('ordered product sales') || h.includes('销售额') || h.includes('销售'));
    if (!hasDate) { result.warnings.push('• 缺少"日期(Date)"列'); result.severity = 'error'; }
    if (!hasSessions) result.warnings.push('• 缺少"会话数(Sessions)"列');
    if (!hasCVR) result.warnings.push('• 缺少"转化率(Unit Session %)"列');
    if (!hasSales) result.warnings.push('• 缺少"销售额(Ordered Product Sales)"列');
    if (data.rows.length < 3) result.warnings.push('• 数据行数太少');
    const dataType = detectDataType(data);
    if (dataType === 'byAsin') {
      const uniqueAsins = new Set(data.rows.map(r =>
        findField(r, ['ASIN', '(Parent ASIN)', 'Parent ASIN', 'asin'])));
      if (uniqueAsins.size > 50)
        result.warnings.push('• 检测到' + uniqueAsins.size + '个ASIN，数据量较大');
    }
  } else if (type === 'ads' || type.startsWith('beforeAds') || type.startsWith('afterAds')) {
    const hasImpressions = headers.some(h => h.includes('impression') || h.includes('曝光'));
    const hasClicks = headers.some(h => h.includes('click') || h.includes('点击'));
    if (!hasImpressions && !hasClicks) {
      result.warnings.push('• 这可能不是广告报表（缺少曝光/点击字段）');
      result.severity = 'error';
    }
  }
  return result;
}

function groupByAsin(data) {

  data.rows.forEach(row => {

// ========== 六、产品KPI计算引擎 ==========
function calcProductKPI(asinInfo) {
  const rows = asinInfo.rows;
  if (!rows || !rows.length) return null;
  const sorted = [...rows].sort((a, b) =>
    (findField(a, ['Date','日期']) || '').localeCompare(findField(b, ['Date','日期']) || ''));

  const sessions = sorted.reduce((s,r) => s + getNum(r,'Sessions','会话'), 0);
  const pv = sorted.reduce((s,r) => s + getNum(r,'Page Views','页面浏览'), 0);
  const cvr = sorted.reduce((s,r) =>
    s + getNum(r,'Unit Session Percentage','Unit Session','转化率'), 0) / sorted.length;
  const sales = sorted.reduce((s,r) => s + getNum(r,'Ordered Product Sales','销售额'), 0);
  const units = sorted.reduce((s,r) => s + getNum(r,'Units Ordered','订购'), 0);
  const pvPerSession = sessions ? pv / sessions : 0;

  // 环比趋势：前后半段对比
  const mid = Math.floor(sorted.length / 2);
  const firstHalf = sorted.slice(0, mid), secondHalf = sorted.slice(mid);
  let sessionsChg = 0, cvrChg = 0, salesChg = 0;
  if (firstHalf.length && secondHalf.length) {
    const s1 = firstHalf.reduce((s,r) => s + getNum(r,'Sessions','会话'), 0);
    const s2 = secondHalf.reduce((s,r) => s + getNum(r,'Sessions','会话'), 0);
    sessionsChg = s1 ? ((s2 - s1) / s1 * 100) : 0;
    const cvr1 = firstHalf.reduce((s,r) =>
      s + getNum(r,'Unit Session Percentage','Unit Session','转化率'), 0) / firstHalf.length;
    const cvr2 = secondHalf.reduce((s,r) =>
      s + getNum(r,'Unit Session Percentage','Unit Session','转化率'), 0) / secondHalf.length;
    cvrChg = cvr1 ? ((cvr2 - cvr1) / cvr1 * 100) : 0;
    const sales1 = firstHalf.reduce((s,r) => s + getNum(r,'Ordered Product Sales','销售额'), 0);
    const sales2 = secondHalf.reduce((s,r) => s + getNum(r,'Ordered Product Sales','销售额'), 0);
    salesChg = sales1 ? ((sales2 - sales1) / sales1 * 100) : 0;
  }

  // 健康度评估算法
  let health = 'bad';
  if (cvr >= 12 && pvPerSession >= 2.5) health = 'good';
  else if (cvr >= 8 || (cvr >= 6 && pvPerSession >= 2.5)) health = 'warn';

  return { sessions, pv, cvr, sales, units, pvPerSession, health,
           title: asinInfo.title, asin: asinInfo.asin,
           sessionsChg, cvrChg, salesChg };
}

function findField(row, candidates) {
  for (const c of candidates) {
    for (const key of Object.keys(row)) {
      if (key.toLowerCase().includes(c.toLowerCase())) return row[key];
    }
  }
  return null;
}
function getNum(row, ...candidates) {
  const val = findField(row, ...candidates);
  if (!val) return 0;
  return parseFloat(String(val).replace(/[^0-9.\-]/g, '')) || 0;
}

// ========== 八、CSV解析器（核心数据输入层） ==========
function parseCSV(text) {
  if (text.charCodeAt(0) === 0xFEFF) text = text.slice(1); // 移除BOM
  const lines = smartSplitLines(text);
  if (lines.length < 2) throw new Error('数据行数不足');

  // 自动查找表头行（处理带说明行的导出文件）
  let headerIdx = 0;
  for (let i = 0; i < Math.min(8, lines.length); i++) {
    const line = lines[i].toLowerCase();
    if (line.includes('date') || line.includes('日期') || line.includes('sessions') ||
        line.includes('impressions') || line.includes('asin') || line.includes('title')) {
      headerIdx = i; break;
    }
    if (i === Math.min(8, lines.length) - 1) headerIdx = 0;
  }

  const headers = parseCSVLine(lines[headerIdx]);
  if (headers.length < 2) throw new Error('无法识别CSV表头');

  const rows = [];
  for (let i = headerIdx + 1; i < lines.length; i++) {
    const line = lines[i].trim();
    if (!line) continue;
    const vals = parseCSVLine(line);
    if (vals.length >= 2 && vals.some(v => v.trim() !== '')) {
      const row = {};
      headers.forEach((h, idx) => { row[h.trim()] = (vals[idx] || '').trim(); });
      rows.push(row);
    }
  }
  if (rows.length === 0) throw new Error('解析到0行数据');
  return { headers: headers.map(h => h.trim()), rows };
}

function smartSplitLines(text) {
  const lines = []; let current = ''; let inQuotes = false;
  for (let i = 0; i < text.length; i++) {
    const ch = text[i];
    if (ch === '"') { inQuotes = !inQuotes; current += ch; }
    else if ((ch === '\n' || ch === '\r') && !inQuotes) {
      if (current.trim()) lines.push(current); current = '';
      if (ch === '\r' && i + 1 < text.length && text[i + 1] === '\n') i++;
    } else { current += ch; }
  }
  if (current.trim()) lines.push(current);
  return lines;
}

function parseCSVLine(line) {
  const result = []; let current = ''; let inQuotes = false;
  for (let i = 0; i < line.length; i++) {
    const ch = line[i];
    if (ch === '"') {
      if (inQuotes && i + 1 < line.length && line[i + 1] === '"')
        { current += '"'; i++; } else inQuotes = !inQuotes;
    } else if (ch === ',' && !inQuotes) { result.push(current); current = ''; }
    else { current += ch; }
  }
  result.push(current);
  return result;
}

// 多编码读取支持（UTF-8 / GBK / Shift_JIS）
function readFileWithEncoding(file) {
  return new Promise((resolve, reject) => {
    const reader = new FileReader();
    reader.onload = function(e) {
      const text = e.target.result;
      const garbled = (text.match(/\ufffd/g) || []).length;
      if (garbled > text.length * 0.05 && text.length > 100) {
        const reader2 = new FileReader();
        reader2.onload = function(e2) { resolve(e2.target.result); };
        reader2.onerror = () => resolve(text);
        reader2.readAsText(file, 'Shift_JIS'); // 日文站点兼容
      } else { resolve(text); }
    };
    reader.onerror = () => reject(new Error('文件读取失败'));
    reader.readAsText(file, 'UTF-8');
  });
}

// ========== 九、文件上传与数据处理控制器 ==========
function handleFileUpload(input, type) {
  const file = input.files[0]; if (!file) return;

  // 文件大小限制检查
  if (file.size > 10 * 1024 * 1024) {
    alert('⚠️ 文件过大（超过10MB），请检查是否选错了文件'); return;
  }

  // 文件格式检查
  if (!file.name.toLowerCase().endsWith('.csv') &&
      !file.name.toLowerCase().endsWith('.tsv') &&
      !file.name.toLowerCase().endsWith('.txt')) {
    alert('⚠️ 请上传 CSV 格式文件\n\n当前文件：' + file.name);
    return;
  }

  readFileWithEncoding(file).then(text => {
    try {
      const data = parseCSV(text);
      if (data.rows.length === 0) {
        alert('⚠️ 文件解析成功但没有数据行'); return;
      }

      // 执行数据校验
      const validation = validateUploadedData(data, type);
      if (validation.warnings.length) {
        const warnMsg = validation.warnings.join('\n');
        if (validation.severity === 'error') {
          alert('❌ 数据格式不匹配\n\n' + warnMsg); return;
        }
        if (!confirm('⚠️ 数据校验发现问题：\n\n' + warnMsg + '\n\n是否继续导入？')) return;
      }

      // 根据数据类型分发存储
      if (type === 'business') {
        appState.bizData = data;
        appState.dataType = detectDataType(data);
        if (appState.dataType === 'byAsin') {
          appState.asinMap = groupByAsin(data);
          appState.selectedAsins = Object.keys(appState.asinMap);
        }
        showDataOverview('business', data);
        renderDashboard(data);
        if (appState.dataType === 'byAsin') { renderProductOverview(); updateProductSelector(); }
      } else if (type === 'ads') {
        appState.adsData = data;
        if (appState.dataType === 'byAsin') appState.asinAdsMap = groupAdsByAsin(data);
        showDataOverview('ads', data); renderAdsData(data);

      // 优化对比数据
      } else if (type === 'beforeBiz') { appState.beforeData = data; updateCompareStatus('before', data);
        if (appState.afterData) renderCompare(appState.beforeData, appState.afterData);
      } else if (type === 'afterBiz') { appState.afterData = data; updateCompareStatus('after', data);
        if (appState.beforeData) renderCompare(appState.beforeData, appState.afterData);
      } else if (type === 'beforeAds') { appState.beforeAdsData = data; updateCompareStatus('before', data, true);
        if (appState.afterData) renderCompare(appState.beforeData, appState.afterData);
      } else if (type === 'afterAds') { appState.afterAdsData = data; updateCompareStatus('after', data, true);
        if (appState.beforeData) renderCompare(appState.beforeData, appState.afterData);

      // 同比分析数据
      } else if (type === 'yoyLastBiz') { appState.yoyLastBiz = data; updateYoyUploadStatus('lastBiz', data);
        if (appState.yoyThisBiz) renderYoyAnalysis();
      } else if (type === 'yoyThisBiz') { appState.yoyThisBiz = data; updateYoyUploadStatus('thisBiz', data);
        if (appState.yoyLastBiz) renderYoyAnalysis();
      } else if (type === 'yoyLastAds') { appState.yoyLastAds = data; updateYoyUploadStatus('lastAds', data);
        if (appState.yoyThisAds) renderYoyAnalysis();
      } else if (type === 'yoyThisAds') { appState.yoyThisAds = data; updateYoyUploadStatus('thisAds', data);
        if (appState.yoyLastAds) renderYoyAnalysis();
      }

      renderQoqAnalysis(); renderBenchmarks(); saveToStorage();

    } catch(err) { alert('❌ 文件解析失败：' + err.message); }
  }).catch(err => { alert('❌ 文件读取失败：' + err.message); });
}
